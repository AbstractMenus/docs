import { getCollection } from "astro:content";

export const entries = (await getCollection("changelog"))
  .sort((a, b) => +b.data.date - +a.data.date);

Subscribe via [RSS](/docs/feed.xml).

{entries.map((entry) => (
  <article style="border-top: 1px solid var(--sl-color-hairline); padding: 1.5rem 0;">
    <header>
      <h2 style="margin: 0;">{entry.data.title}</h2>
      <p style="margin: 0.25rem 0 1rem; color: var(--sl-color-gray-3); font-size: 0.875rem;">
        <time datetime={entry.data.date.toISOString()}>
          {entry.data.date.toLocaleDateString("en-US", { year: "numeric", month: "long", day: "numeric" })}
        </time>
        {entry.data.version ? <> &middot; <code>{entry.data.version}</code></> : null}
      </p>
    </header>
    {entry.data.summary && <p>{entry.data.summary}</p>}
    {entry.data.tags.length > 0 && (
      <p style="font-size: 0.875rem; color: var(--sl-color-gray-3);">
        {entry.data.tags.map((tag) => <code key={tag} style="margin-right: 0.5rem;">#{tag}</code>)}
      </p>
    )}
  </article>
))}
